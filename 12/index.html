Calculate factorial both recursively and non recursively.

  // recursively
  function factorialR(n){
    if (n === 1) return 1;  
    return n * factorialR(n-1);
  }  
  console.log(factorialR(4));

  // non recursively
  function factorialNR(n){
    let total = 1;
    for (var i=n; i>=1; i--){
      total *= i;
    }
    return total;
  }
  console.log(factorialNR(4));
