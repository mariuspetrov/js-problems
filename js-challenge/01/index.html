Function that takes a string as an argument and returns the frequency of the letters inside the string.

const str = 'Lorem ipsum dolor sit amet!';
const obj = {};

function lettersFreq(str){
  for (let el of str) {
    obj[el] = obj[el] ? ++obj[el] : 1;
  }
  return obj;
}

console.log(lettersFreq(str));
